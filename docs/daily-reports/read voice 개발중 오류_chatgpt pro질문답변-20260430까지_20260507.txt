A. 새 파일 생성

app/api/classify-image/route.ts 파일을 새로 만드세요.

import { NextRequest, NextResponse } from "next/server"
import { classifyImage } from "@/modules/ocr/gemini"
import { MAX_FILE_SIZE } from "@/modules/ocr"

export const runtime = "nodejs"

export async function POST(req: NextRequest) {
  try {
    const contentType = req.headers.get("content-type") ?? ""

    if (!contentType.includes("multipart/form-data")) {
      return NextResponse.json(
        { error: "Invalid content type" },
        { status: 415 }
      )
    }

    const formData = await req.formData()
    const file = formData.get("file")

    if (!file || !(file instanceof File)) {
      return NextResponse.json(
        { error: "No file provided" },
        { status: 400 }
      )
    }

    if (!file.type.startsWith("image/")) {
      return NextResponse.json(
        { error: "이미지 파일만 자동 분류할 수 있습니다." },
        { status: 400 }
      )
    }

    if (file.size > MAX_FILE_SIZE) {
      return NextResponse.json(
        { error: "파일 크기는 10MB 이하여야 합니다." },
        { status: 400 }
      )
    }

    const arrayBuffer = await file.arrayBuffer()
    const buffer = Buffer.from(arrayBuffer)

    const classification = await classifyImage(buffer, file.name)

    return NextResponse.json({ classification })
  } catch (err) {
    const message =
      err instanceof Error ? err.message : "이미지 자동 분류에 실패했습니다."

    console.error("[API classify-image]", err)

    return NextResponse.json(
      { error: message },
      { status: 500 }
    )
  }
}

현재 /api/ocr는 이미지/PDF 파일을 받아 모델 유효성 검사 후 이미지 분석 또는 PDF OCR로 넘깁니다. PDF 쪽은 fs, child_process, Python subprocess, Ollama 호출을 사용하므로 Node 런타임을 명시하는 편이 안전합니다.

3단계. 실행 테스트

먼저 Ollama 확인:

ollama list
curl.exe http://localhost:11434/api/tags

그 다음 Next 개발 서버 재시작:

cd C:\Users\tara0\readvoice-mvp
npm run dev

브라우저를 열기 전에 API부터 직접 테스트하세요.

curl.exe -s -X POST "http://localhost:3000/api/classify-image" `
  -F "file=@public\ReadVoice_Upload\test_ocr.png;type=image/png"

정상이라면 이런 형태가 나와야 합니다.

{"classification":"document"}

그 다음 이미지 OCR 테스트:

curl.exe -s -X POST "http://localhost:3000/api/ocr" `
  -F "file=@public\ReadVoice_Upload\test_ocr.png;type=image/png" `
  -F "model=gemma4:e4b"

판정 기준은 다음과 같습니다.

/api/classify-image 성공 + /api/ocr 성공 + 브라우저 실패
→ FileUpload 이벤트/TTS/모델 선택 흐름 문제

/api/classify-image 실패
→ 이미지 자동분류 서버 API 또는 Ollama 연결 문제

/api/ocr 실패
→ Vision 모델, Ollama, Python/PDF/OCR 서버 처리 문제

업로드된 개발 보고서에도 “PowerShell에서는 성공하지만 브라우저에서 PDF OCR 실패, Node.js subprocess에서 Ollama PATH를 못 찾는 것으로 추정”이라는 이슈가 남아 있습니다. 따라서 PDF까지 함께 고치려면 브라우저에서 직접 처리하지 말고 서버 API 라우트 중심으로 정리하는 방향이 맞습니다.
